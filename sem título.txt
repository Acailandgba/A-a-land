<script>
    function finalizarPedido() {
        let tamanho = document.getElementById("tamanho").value;
        let complementos = [];
        document.querySelectorAll("input[type=checkbox]:checked").forEach(el => complementos.push(el.value));
        let pagamento = document.getElementById("pagamento").value;
        let endereco = document.getElementById("endereco").value;
        
        alert(`Pedido:\nTamanho: ${tamanho}\nComplementos: ${complementos.join(", ")}\nForma de Pagamento: ${pagamento}\nEndereço: ${endereco}`);
    }
</script>
